import arcade

WIDTH = 600
HEIGHT = 300

def main():
    global bx, by, vx, vy
    arcade.open_window(WIDTH, HEIGHT, 'Example')
    arcade.set_background_color(arcade.color.BLUE_SAPPHIRE)
    bx, by = WIDTH/2, HEIGHT/2
    vx, vy = 1, 0.5
    arcade.schedule(on_draw, 1/60)
    arcade.run()

def draw_happy_box(x,y):
    arcade.draw_rectangle_filled(x, y, 50, 50, arcade.color.RED_DEVIL)
    arcade.draw_circle_filled(x+7, y+7, 3, arcade.color.BLACK)
    arcade.draw_circle_filled(x-7, y+7, 3, arcade.color.BLACK)
    arcade.draw_arc_filled(x, y, 20, 10, arcade.color.BLACK, 180, 360)

def on_draw(time_step):
    global bx, by, vx, vy
    arcade.start_render()
    draw_happy_box(bx, by)
    bx += vx
    by += vy
    if bx > WIDTH or bx < 0:
        vx *= -1
    if by > WIDTH or by < 0:
        vy *= -1

main()
