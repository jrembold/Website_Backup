def odd_sum(maxnum=10):
    global total
    def add_odd(x):
        global total
        if x % 2 == 1:
            total += x

    total = 0
    for n in range(maxnum+1):
        add_odd(n)
    return total

def circumference(radius):

	return pi * radius ** 2

def pie_area(radius, angle):
	return area(radius) * angle / (2 * pi)
