import matplotlib.pyplot as plt

f = plt.figure()
ax = f.add_subplot(211)
ax2 = f.add_subplot(212)

ax.plot([7,6,4,3,1], '^g--')
ax2.plot([1,2,3,7,6,5],[3,10,7,6,8,5], '*r-',
    linewidth=4, markersize=15, color='blue')
plt.show()
