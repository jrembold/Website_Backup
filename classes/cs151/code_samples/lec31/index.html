import matplotlib.pyplot as plt
plt.style.use('fivethirtyeight')

f = plt.figure(figsize=(10,5))
ax1 = f.add_subplot(121)
ax2 = f.add_subplot(122)

ax1.plot([1,2,3,4,5], 'gd')
ax2.plot([2,4,9,16,25], 'bs--')

ax1.set_xlabel('Axis 1 X')
ax1.set_ylabel('Axis 1 Y')

ax2.set_xlabel('Axis 2 X')
ax2.set_ylabel('Axis 2 Y')

ax1.set_title('Title of Plot 1')
ax2.set_title('Title of Plot 2')

f.subplots.adjust(hspace=.5, left=.2, bottom=.2)

f.suptitle('Two lovely plots!', fontsize=20, color='red')

ax1.set_ylim(0,10)
ax1.set_yticks(range(11))

ax2.set_xticks([0,1,1.5,2,3])
ax2.set_xticklabels(['A', 'B', 'C', 'D', 'E'])

plt.show()

import matplotlib.pyplot as plt

fig = plt.figure()

ax1 = fig.add_subplot(111)
ax2 = ax1.twinx()

ax1.plot([1,2,3,4,5], 'g')
ax2.plot([1,4,9,16,25], 'r')

ax1.set_ylabel('Axis 1 Y Label')
ax2.set_ylabel('Axis 2 Y Label')
ax1.set_xlabel('Shared X Axis')

plt.show()
