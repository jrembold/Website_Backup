s = input('Enter a string of decimal numbers'
        ' separated by commas: ')

curr_num = ''
running_total = 0
for c in s:
    if c == ',':
        running_total = running_total + float(curr_num)
        curr_num = ''
    else:
        curr_num = curr_num + c
running_total = running_total + float(curr_num)
print(running_total)
