import arcade

class Example(arcade.Window):
    def __init__(self, width, height, title):
        arcade.Window.__init__(self, width, height, title)
        arcade.set_background_color(arcade.color.BLUE_YONDER)

        self.width = width
        self.height = height

        self.x = self.width/2
        self.y = self.height/2
        self.r = 100
        self.color = arcade.color.BOYSENBERRY
        self.xdir = 1
        self.ydir = 2

        arcade.run()

    def on_draw(self):
        arcade.start_render()
        arcade.draw_circle_filled(self.x, self.y, self.r, self.color)

    def on_update(self, dt):
        self.x += self.xdir
        self.y += self.ydir

        if (self.x+self.r) > self.width or (self.x-self.r)<0:
            self.xdir *= -1
        if (self.y+self.r) > self.height or (self.y-self.r)<0:
            self.ydir *= -1

    def on_key_press(self, key, mod):
        if key == arcade.key.ESCAPE:
            arcade.close_window()
        elif key == arcade.key.SPACE:
            if self.color == arcade.color.BOYSENBERRY:
                self.color = arcade.color.BRIGHT_GREEN
            else:
                self.color = arcade.color.BOYSENBERRY

    def on_mouse_press(self, x, y, button, mod):
        if button == arcade.MOUSE_BUTTON_LEFT:
            arcade.set_background_color(arcade.color.GUPPIE_GREEN)


Example(500, 500, 'Arcade Example')
