pi = 3.14159

def circumference(radius):
    return 2 * pi * radius

def area(radius):
    return pi * radius ** 2

def pie_area(radius, angle):
    return area(radius) * angle / (2 * pi)

if __name__ == '__main__':
    # This code will not be run upon import!
    print(area(5))
