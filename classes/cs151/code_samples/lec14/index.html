def mult_w_loop(A,B):
    total = 0
    for i in range(B):
        total += A
    return total

def mult_w_rec(A,B):
    if B == 1:
        return A
    else:
        return A + mult_w_rec(A,B-1)

def factorial(n):
    if n == 1:
        return 1
    else:
        return n * factorial(n-1)
