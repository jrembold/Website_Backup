class Student:
   def __init__(self, name, age):
       self.name = name
       self.age = age
       self.is_human = True


s1 = Student("Bob", 21)
s2 = Student("Jill", 18)
s2.is_human = False
s1.age += 1

for student in [s1, s2]:
   print(
       f'{student.name} is {student.age} and {"is" if student.is_human else "is not"} human.'
   )

class Coordinate:
    def __init__(self, x, y):
        self.x = x
        self.y = y
        self.turkey = True

    def dist2origin(self):
        d = (self.x ** 2 + self.y ** 2) ** 0.5
        return d

    def distance2coord(self, othercoord):
        diffx = self.x - othercoord.x
        diffy = self.y - othercoord.y
        d = (diffx ** 2 + diffy ** 2) ** 0.5
        return d


C1 = Coordinate(1, 2)
C2 = Coordinate(5, 6)

print(C1.dist2origin())
print(Coordinate.dist2origin(C2))
print(C1.distance2coord(C2))
print(C2.distance2coord(C1))
