B = ('samurai', 'pirate', 'ninja')

print('\nMethod 1\n----------')
for i in range(len(B)):
    print(i, ':', B[i])

print('\nMethod 2\n----------')
i = 0
for s in B:
    print(i, ':', s)
    i += 1

print('\nMethod 3\n----------')
for index, value in enumerate(B):
    print(index, ':', value)
