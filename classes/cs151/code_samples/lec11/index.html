def year_savings(init_amt, int_rate):
    for i in range(12):
        init_amt *= (1+int_rate/12)
    return init_amt

amount = 100
rate = 0
while rate < 1:
    print('Savings:', year_savings(amount, rate))
    rate += 0.05

print('All done!')

def is_prime(num):
    for d in range(2,num):
        if num % d == 0:
            return False
    return True

for a in range(100):
    for b in range(a):
        if is_prime(a) and is_prime(b) and (a+b == num):
            print(a, 'and', b, 'are both prime and sum to 100!')
