goal = 0
epsilon = 0.01
guess = 1
num_guesses = 1

def f(x):
    return 8 * x**2 - 3

while abs(f(guess) - goal) > epsilon:
    numerator = f(guess)
    denom = 16 * guess
    guess -= numerator / denom
    num_guesses += 1
print('Solution:', guess)
print('It took', num_guesses, 'guesses!')

def f(x):
    return (g+3)**3 + 5*g**2 - 15*g - 200

def df(x):
    return 3*(g+3)**2 + 10*g - 15
while abs(f(g) - 0) > epsilon:
    num = f(g)
    deriv = df(g)
    g -= num/deriv
