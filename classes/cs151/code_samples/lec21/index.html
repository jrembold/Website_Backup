    return x ** 2

def f2(x):
    return x // 2

def f3(x):
    if x > 0.5:
        return 1
    else:
        return 0

L = [f1,f2,f3]
A = [1,2,3,4,5]

for num in A:
    print('Number:',num)
    for f in L:
        print(f(num))

def apply_func_2_list(func, list_):
    L = []
    for value in list_:
        L.append(func(value))
    return L

def f1(x):
    return x ** 2

def f2(x):
    return x // 2

A = [1,2,3,4,5]
print('Function 1 applied to A': apply_func_2_list(f1, A))
print('Function 2 applied to A': apply_func_2_list(f2, A))
