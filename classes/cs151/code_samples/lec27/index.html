class Animal:
    def __init__(self, name, age):
        self.name = name
        self._age = age # Protected
        self.__secret = 'Afraid of water' # Private

    def __repr__(self):
        return f'Animal({self.name}, {self._age})'

    def get_name(self):
        return self.name

    def change_name(self, new_name):
        self.name = new_name

    def get_age(self):
        return self._age

    def set_age(self, new_age):
        self._age = new_age

    def get_secret(self):
        return self.__secret

A = Animal('Barry', 43)
print(A.name) # Can access but not recommended
print(A.get_name()) # Better to use getter
print(A.get_age())
A.set_age(34)
print(A.get_age())
print(A.get_secret()) # Can access this way
#print(A.__secret) # This will give an error

class Human(Animal):
    def talk(self):     # defining new method
        print("I am a human!")

    def __repr__(self): # override method
        return f'Human({self.name},{self._age})'

H = Human('Walt', 23) # Need same parameters as Animal
H.talk()
print(H.get_age())
print(H.get_secret()) # Private attributes don't get passed on thru inheritance
