class MyBestClass:
    a_class_variable = 'Hello!'

    def __init__(self):
        self.a_instance_variable = 'Hi!'

print(MyBestClass.a_class_variable)
A = MyBestClass()
print(A.a_class_variable)

# Changing things:
B = MyBestClass()
A.a_class_variable = 'Oh hi!'
print(A.a_class_variable, B.a_class_variable)

class Checker:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def check_move(self):
        pass

    def move(self):
        pass

    def jump(self):
        pass

class RedChecker(Checker):
    color = 'red'
    count = 0

    def __init__(self, x, y):
        Checker.__init__(self, x, y)
        self.direction = 'Up'
        RedChecker.count += 1

    def __del__(self):
        RedChecker.count -= 1
