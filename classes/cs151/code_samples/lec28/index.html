class Human(object):
    def __init__(self,age):
        self.age = age
        self.legs = 2
        self.eyes = 2

    def __repr__(self):
        return f'Human({self.age})'

class Pirate(Human):
    def talk(self):
        return "Aye matey!"

    def __repr__(self):
        return f'Pirate({self.age})'

class Pegleg(Pirate):
    def __init__(self, age):
        # This works, but is not the recommended way!
        self.age = age
        self.eyes = 2
        self.legs = 1
        self.type_peg = 'oak'

    def __repr__(self):
        return f'Pegleg({self.age})'

    def __str__(self):
        return 'I have one leg!'

class Patch(Pirate):
    def __init__(self, age):
        Pirate.__init__(self, age)
        self.eyes = 1

    def __repr__(self):
        return f'Patch({self.age})'

class Pirate(object):
    def __init__(self, age):
        self.plunders = True
        self.age = age
        self.eats = 'food'

class Zombie(object):
    def __init__(self):
        self.type = 'Undead'
        self.eats = 'brains'

class Black_Pearl_Crew(Pirate, Zombie):
    def __init__(self, age):
        Pirate.__init__(self, age)
        Zombie.__init__(self)

Barbosa = Black_Pearl_Crew(56)
print(f'Barbosa is {Barbosa.age} and eats {Barbosa.eats}')
