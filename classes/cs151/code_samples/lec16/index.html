def rec(s):
    if len(s) <= 1:
        return s.upper()
    elif s[-2] == ' ':
        return rec(s[:-2]) + s[-1].upper()
    else:
        return rec(s[:-1]) + s[-1]

import arcade
import math

def main(size):
    arcade.open_window(size, size, 'TRIANGLES')
    arcade.set_background_color(arcade.color.BLACK)
    arcade.start_render()
    draw_rec_tri(0,0,size)
    arcade.finish_render()
    arcade.run()

def draw_rec_tri(x,y,L):
    if L > 3:
        p1x = x
        p1y = y
        p2x = x+L
        p2y = y
        p3x = x + L * math.cos(60*math.pi/180)
        p3y = y + L * math.sin(60*math.pi/180)
        arcade.draw_triangle_outline(p1x, p1y, p2x, p2y, p3x, p3y, arcade.color.GREEN)

        draw_rec_tri(x, y, L/2)
        draw_rec_tri(x+L/2, y, L/2)
        draw_rec_tri(x + L/2 * math.cos(60*math.pi/180),
            y + L/2 * math.sin(60*math.pi/180), L/2)

main(600)

import math

# Amplitude of 20
def wave(x):
    return int(20 * math.sin(x/5) + 20)

def line(x):
    dot_index = wave(x)
    pre = '-'* (dot_index-1)
    return pre + ':)' 

f = open('Sine.txt', 'w')
for x in range(100):
    f.write(line(x) + '\n')
f.close()
